  5 /** ProtoPopup.Dialog is based on ProtoPopup and ads 'OK' and
  6     'Cancel' buttons to its base object.
  7     @class Creates a ProtoPopup.Dialog object
  8     @constructor
  9     @augments ProtoPopup
 10     @param {STRING} id A unique string identifying a popup
 11     @param {OBJECT} config The configuration object {@link #.config}
 12     @return ProtoPopup.Dialog object
 13     @property {object} config The default configuration inherited from
 14     {@link ProtoPopup#config} augmented with:
 15     <div style="padding-left: 20px">
 16        <b>buttons:</b> An array of button specs objects. The buttons are inserted
 17        into the footer in the order they are specified.<br/><br/>
 18 
 19        <b>A button spec object takes the following keys:</b><br/><br/>
 20 
 21        <b>name:</b> The name will be used to build the ID of the button. Give
 22        a popup ID 'dialog' and a button name 'save', the button's ID will be 'dialog-save-btn'<br/>
 23 
 24        <b>label:</b> The label of the button.<br/>
 25 
 26        <b>onClick:</b> The callback function executed when the button is clicked.<br/>
 27 
 28        <b>giveFocus:</b> A boolean specifying whether the button should be focused when the popup is displayed.<br/>
 29 
 30        <b>backgroundImage</b> {STRING} - CSS property
 31        'background-image' for the button. Defaults to
 32        undefined.<br/>
 33     </div>
 34 */
 35 ProtoPopup.Dialog = Class.create(ProtoPopup, /** @lends ProtoPopup.Dialog.prototype */{
 36     /** @ignore */
 37     initialize : function($super, id, config) {
 38         var _config = {
 39             buttons: []
 40         };
 41         Object.extend(_config, (config || {}));
 42         $super(id, _config);
 43         
 44         // make the buttons
 45         this.config.buttons.each(function(spec) {
 46             var btn  = this.makeButton(spec.name, spec.label);
 47             this.footer.insert(btn).show();
 48             if (spec.giveFocus) { this.onShow.push(function() {btn.focus()}) }
 49 
 50             // attach handler
 51             var clickHandler = function(e) {
 52                 this.hide();
 53                 spec.onClick();
 54             }.bind(this);
 55 
 56             Event.observe(btn, 'click', clickHandler);
 57 
 58         }.bind(this));
 59     }
 60 });

 62 /**
 63    Class method returning (maybe first create) a draggable popup DIV
 64    for confirm dialogs.   Given the same id argument returns the same popup
 65    object, following the singleton pattern. See the example of
 66    the base class' {@link ProtoPopup.get}
 67    @function
 68    @param {STRING} id The name of the popup used to build its ID.
 69    @param {OBJECT} config The config object, see {@link #.config}.
 70    @return The initialized and draggable popup.
 71 */
 72 ProtoPopup.Dialog.get = ProtoPopup.makeGetFor(function(id, config) {
 73     return new ProtoPopup.Dialog(id, config)
 74 });
