1 /** @fileoverview ProtoPopup and its subclasses ProtoPopup.Alert and
  2     ProtoPopup.Confirm provide a light-weight solution for info, alert
  3     and confirm dialogs based on Prototype and Scriptaculous.
  4     @author <a href="mailto:bs@cms-schulze.de">Bodo Schulze</a>
  5     @version 0.1
  6     @license BSD-like
  7     @www <a href="http://dev.cms-schulze.de>ProtoPopup</a>
  8 
  9    <h4 style="margin-bottom:0em">1. Features</h4>
 10 
 11    <ul>
 12    <li>Multiple popups may be open at the same time.</li>
 13 
 14    <li>Position memory: Each popup remembers its position relative to
 15      the viewport across showing / hiding: The popups are draggable,
 16      but have position 'fixed' in FF2+, IE7+, Webkit.</li>
 17 
 18    <li>Content sections: Popups have three section DIVs: 'header',
 19      'body', 'footer'. They can be filled with content separately.</li>
 20 
 21    <li>Flexible content definition: The section's content can be plain
 22      text, HTML, a DOM element or any kind of object having a
 23      toString() method (internally uses Prototype's Element.update()).</li>
 24 
 25    <li>Append mode: Each content section may be configured as persisting
 26      its content from the previous call, switching the section into
 27      append mode.</li>
 28 
 29    <li>Three flavors:
 30      <ul>
 31          <li>buttonless info box (hidden via ESC) implemented in the
 32            ProtoPopup base class</li>
 33 
 34          <li>alert-like info box with 'Close' button implemented in
 35            ProtoPopup.Alert</li>
 36 
 37          <li>confirm-like prompt with 'OK' and 'Cancel' buttons
 38            implemented in ProtoPopup.Confirm</li>
 39      </ul></li>
 40    </ul>
 41 
 42 */

 48 /** ProtoPopup provides a button-less info box with header, body and
 49     footer fields.  Since it has no buttons, closing is done via the
 50     ESC key.
 51     @class
 52     @constructor
 53     @requires prototype-1.6.3.js -- the wellknown Ajax library, as well as Scriptaculous' effects.js and dragdrop.js
 54     @param {STRING} id A unique string identifying a popup
 55     @param {OBJECT} config The configuration object
 56     @return ProtoPopup object
 57     @type object
 58 
 59 
 60 
 61     @example
 62     var info = new ProtoPopup('info', {
 63         header     : new Element('div').update('Info header'),
 64         footer     : 'Info footer',
 65         appendBody : true
 66     });
 67 
 68     // Show the first info
 69     info.setHtml({body : 'Info 1'}).show();
 70 
 71     // Append a second info in bold
 72     info.setHtml({body : new Element('strong').update('Info 2')}).show();
 73 */
 74 var ProtoPopup = Class.create(/** @lends ProtoPopup.prototype */{
 75 
 76     /** @ignore */
 77     initialize : function (id, config) {
 78 
 79         /**
 80            The ID of the popup's object and DIV.
 81          */
 82         this.id = id;
 83 
 84         /**
 85            The default configuration<br/>
 86 
 87            <div style="padding-left: 20px">
 88 
 89            <b>hideOnEscape</b> {BOOL} - If true hides the popup when
 90            pressing the ESC key. Defaults to true.<br/>
 91 
 92            <b>centerOnCreation</b> {BOOL} - Center the popup relative
 93            to the viewport when creating it. Lateron it will be placed
 94            where the user drags it to. The position is remembered
 95            across hiding/showing until a full page reload
 96            occurs. Defaults to true.<br/>
 97 
 98            <b>appendHeader</b> {BOOL} - If true, calls to {@link #setHtml}
 99            will append to the popup's header section.  Defaults to true.<br/>
100 
101            <b>appendBody</b> {BOOL} - If true, calls to {@link #setHtml}
102            will append to the popup's body section.  Defaults to true.<br/>
103 
104            <b>appendFooter</b> {BOOL} - If true, calls to {@link #setHtml}
105            will append to the popup's footer section. Defaults to true.<br/>
106 
107            <b>header</b> {STRING} - The initial value of the header
108            section. Defaults to undefined.<br/>
109 
110            <b>body</b> {STRING} - The initial value of the footer
111            section. Defaults to undefined.<br/>
112 
113            <b>footer</b> {STRING} - The initial value of the body
114            section. Defaults to undefined.<br/>
115 
116            <b>width</b> {STRING} - The width of the popup. Defaults to
117            300px.<br/>
118 
119            <b>cancelIconSrc</b> {STRING} - The src URL of the cancel
120            icon. Defaults to 'images/cancel.png'<br/>
121 
122            <b>zIndex</b> {NUMBER} - The z-index of the popup. Defaults
123            to 0.<br/>
124 
125            </div>
126 
127         */
128         this.config = {
129             hideOnEscape     : true,
130             centerOnCreation : true,            
131             appendHeader     : false,
132             appendBody       : false,
133             appendFooter     : false,
134             header           : undefined,
135             body             : undefined,
136             footer           : undefined,
137             width            : '300px',
138             cancelIconSrc    : 'images/cancel.png',
139             zIndex           : 0
140         };
141 
142         // merge in the config
143         Object.extend(this.config, (config || {}));
144 
145         // maybe attach keyUp handler to hide via ESC
146         if (this.config.hideOnEscape) {
147             Event.observe(document, 'keyup', this.hideOnEscape);
148         }
149 
150         var popup = new Element('div', {id : id, 'class' : 'proto-popup'})
151             .setStyle({width : this.config.width, zIndex : this.config.zIndex}).hide();
152             
153         /**
154            The popup div having {@link #id} as its ID, 'proto-popup'
155            as its class attribute.
156          */
157         this.popup = popup;
158 
159         // insert the cancel icon and attach click handler
160         popup.insert(new Element('img', {
161             src     : this.config.cancelIconSrc,
162             'class' : 'proto-popup-cancel'
163         }).observe('click', function(e) { popup.hide(); Event.stop(e) }));;
164 
165         /**
166            Array holding the popup's section names 'header', 'body' and 'footer'.
167          */
168         this.sections = ['header', 'body', 'footer'];
169 
170         /**
171            The popup's 'header' section is the popup's first DOM
172            child. It's ID is "id+'-header'", it's class attribute
173            'proto-popup-header'.   It's content is set via
174            {@link #setHtml}.
175 
176            @name ProtoPopup.prototype.header
177 
178          */
179         /**
180            The popup's 'body' section is the popup's first DOM
181            child. It's ID is "id+'-body'", it's class attribute
182            'proto-popup-body'.  It's content is set via
183            {@link #setHtml}.
184 
185            @name ProtoPopup.prototype.body
186 
187          */
188         /**
189            The popup's 'footer' section is the popup's first DOM
190            child. It's ID is "id+'-footer'", it's class attribute
191            'proto-popup-footer'.  It's content is set via
192            {@link #setHtml}.
193 
194            @name ProtoPopup.prototype.footer
195 
196          */
197         this.sections.each(function(section) {
198                 this[section] = new Element('div', {id : id+'-'+section, 'class' : 'proto-popup-'+section}).hide();
199                 popup.insert(this[section]);
200         }.bind(this));
201 
202         // append to document body
203         document.body.appendChild(this.popup);
204 
205         // make it draggable
206         new Draggable(id);
207         
208         // normally it's position:fixed, but IE6- does not understand it
209         if (Prototype.Browser.IEVersion < 7) {
210             popup.setStyle({position: 'absolute'});
211         }
212 
213         // set HTML
214         this.setHtml(this.config);
215 
216         // remember us
217         ProtoPopup.id2obj[id] = this;
218 
219         /**
220            Array of functions that will be executed at the end of the
221            popup initialization.
222          */
223         this.onShow = [];
224     },
225 
226     /**
227        Fired when the ESC key is pressed and the config option
228        hideOnEscape is true (which is the default).
229        @event
230      */
231     hideOnEscape : function(event) {
232         if (event.keyCode == 27) { // KEY_ESC
233             Object.keys(ProtoPopup.id2obj).each(function(pp) {
234                     if (ProtoPopup.id2obj[pp].config.hideOnEscape) {
235                         ProtoPopup.id2obj[pp].hide();
236                     }
237             });
238         }
239 
240     }.bindAsEventListener(this),
241 
242     /**
243        Chainable instance method to set the HTML in any of the three popup
244        sections 'header', 'body' or 'footer'.
245 
246        @param config An object whose keys represent said three
247        sections.  The corresponding values can be plain text, HTML, a
248        DOM node or any object having a toHTML() method.
249 
250        @example
251        var info = new ProtoPopup('info');
252        info.setHtml({header : 'Info header', body : 'My Info'});
253     */
254     setHtml : function(config) {
255         this.sections.each(function(section) {
256                 if (config && config[section] != undefined) {
257                 if (!this.config['append'+section.capitalize()]) {
258                     this[section].innerHTML = '';
259                     
260                 }
261                 this[section].insert(config[section]).show();
262             }
263         }.bind(this));
264         return this;
265     },
266 
267     /**
268        Chainable instance method to show the popup.
269        @example
270        var info = new ProtoPopup('info', {header : 'Info header'});
271        info.setHtml({body : 'My Info'}).show();
272     */
273     show : function() {
274         this.popup.show();
275         if (this.config.centerOnCreation) {
276             this.centerIt();
277             this.config.centerOnCreation = false;
278         }
279         this.onShow.each(function(f) {
280             if (Object.isFunction(f)) f.defer();
281         });
282         return this;
283     },
284 
285     /**
286        Chainable instance method to hide the popup.
287        @example
288        var info = new ProtoPopup('info', {header : 'Info header'});
289        info.setHtml({body : 'My Info'}).show();
290 
291        // hide 3 seconds later after showing it
292        setTimeout(function() {pinfo.hide()}, 3000);
293 
294     */
295     hide : function() {
296         this.popup.hide();
297         return this;
298     },
299 
300     /**
301        Chainable instance method to center the popup horizontally and
302        vertically on the viewport.  Internally called by {@link #show}.
303     */
304     centerIt : function() {
305         windowDim = document.viewport.getDimensions();
306         popupDim  = this.popup.getDimensions();
307 
308         centerX = Math.round(windowDim.width / 2) 
309             - (popupDim.width  / 2) + 'px';
310 
311         centerY = Math.round(windowDim.height/ 2) 
312             - (popupDim.height / 2) + 'px';
313 
314         this.popup.setStyle({left: centerX, top: centerY});
315 
316         return this;
317     },
318 
319     /**
320        Non-chainable instance method to create and return a button element.
321        This method is not used by ProtoPopup, but by its child classes
322        {@link ProtoPopup.Alert} and {@link ProtoPopup.Confirm}.<br><br>
323 
324        It's CSS class is "'proto-popup-'+name+'-btn'".
325 
326     */
327     makeButton : function(name) {
328         return new Element('input', {
329             type    : 'button',
330             value   : this.config[name+'BtnLabel'],
331             'class' : 'proto-popup-'+name+'-btn'
332         });
333    }
334 });

336 /**
337    Class method returning get() methods for ProtoPopup and its
338    derivations.
339    @param newProtoPopup Function returning a constructor making ProtoPopup or
340    one of derivative objects.
341    @return get() method
342 */
343 ProtoPopup.makeGetFor = function(newProtoPopup) {
344     return function(id, config) {
345         var pp;
346         if (pp = ProtoPopup.id2obj[id]) {
347             // we have it
348             pp.setHtml(config);
349             return pp;
350         }
351         // create it
352         pp = newProtoPopup(id, config);
353         return pp;
354     }
355 }
356 
357 /**
358    Class method returning makeFunction() methods for ProtoPopup and its
359    derivations.
360    @param newProtoPopup Function returning a constructor making ProtoPopup or
361    one of derivative objects.
362    @return get() method
363 */
364 ProtoPopup.makeMakeFunction = function(newProtoPopup) {
365     return function(id, config) {
366         var pp;
367         if (Object.isUndefined(pp)) {
368             pp = newProtoPopup(id, config);
369             if (config && config.arg) {
370                 // attach original object
371                 config.arg.callee.obj = pp;
372             }
373         }
374         return function(msg) {
375             pp.setHtml({body : msg}).show();
376         }
377     }
378 }

380 /**
381    Class method returning (maybe first create) a draggable popup DIV
382    for info boxes.  Given the same id argument returns the same popup
383    object, following the singleton pattern.
384    @function
385    @param {STRING} id The ID of the popup.
386    @param {OBJECT} config The {@link #config} object.
387    @return The initialized and draggable popup.
388    @example Make a custom info() method serving as a convenient wrapper.
389 
390    The popup shown by info() will always have the string
391    'Newest Information' in the popup's header section, while the
392    argument passed to info() will be wrapped in a DIV that
393    will be inserted in the popup's body section.
394    
395         // make the custom info function
396         var info = function(msg) {
397             ProtoPopup.get('info', {
398                 header     : 'Info header',
399                 body       : new Element('div').update(msg),
400             });
401         }
402 
403         // first use
404         info('My first info');
405 
406         // second use (using the same popup, showing the same header)
407         info('My second info'); 
408 */
409 ProtoPopup.get = ProtoPopup.makeGetFor(function(id, config) {
410     return new ProtoPopup(id, config)
411 });
412 
413 /**
414    Class method making custom functions accepting one argument that
415    will be inserted in the body section of the underlying ProtoPopup
416    object created behind the scenes.
417    @function
418    @param {STRING} id The ID of the popup.
419    @param {OBJECT} config The {@link #config} object.
420    @return A function accepting one argument to be inserted into the
421    popup's body section.
422    @example Make a custom info() method
423 
424       // make the method
425       var info = ProtoPopup.makeFunction('info', {
426           header : 'Announcement'
427       });
428 
429       // first use
430       info('Forget this info');
431 
432       // second use (using the same popup, showing the same header
433       info('Forget this info, too');
434 */
435 ProtoPopup.makeFunction = ProtoPopup.makeMakeFunction(function(id, config) {
436     return new ProtoPopup(id, config);
437 });
