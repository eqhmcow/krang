1 /** @fileoverview ProtoPopup and its subclasses ProtoPopup.Alert and
  2     ProtoPopup.Confirm provide a light-weight solution for info, alert
  3     and confirm dialogs based on Prototype and Scriptaculous.
  4     @author <a href="mailto:bs@cms-schulze.de">Bodo Schulze</a>
  5     @version 0.4
  6     @license BSD-like
  7     @www <a href="http://protopopup.cms-schulze.de>ProtoPopup</a>
  8 
  9    <h4 style="margin-bottom:0em">1. Features</h4>
 10 
 11    <ul>
 12    <li>Multiple popups may be open at the same time.</li>
 13 
 14    <li>Position memory: Each popup remembers its position relative to
 15      the viewport across showing / hiding: The popups are draggable,
 16      but have position 'fixed' in FF2+, IE7+, Webkit.</li>
 17 
 18    <li>Content sections: Popups have three section DIVs: 'header',
 19      'body', 'footer'. They can be filled with content separately.</li>
 20 
 21    <li>Flexible content definition: The section's content can be plain
 22      text, HTML, a DOM element or any kind of object having a
 23      toString() method (internally uses Prototype's Element.update()).</li>
 24 
 25    <li>Append mode: Each content section may be configured as persisting
 26      its content from the previous call, switching the section into
 27      append mode.</li>
 28 
 29    <li>Four flavors:
 30      <ul>
 31          <li>buttonless info box (hidden via ESC) implemented in the
 32            ProtoPopup base class</li>
 33 
 34          <li>alert-like info box with 'Close' button implemented in
 35            ProtoPopup.Alert</li>
 36 
 37          <li>confirm-like prompt with 'OK' and 'Cancel' buttons
 38            implemented in ProtoPopup.Confirm</li>
 39 
 40          <li>dialogs with arbitrary buttons in header and/or footer section,
 41            with left/right positioning and arbitrary event registration</li>
 42      </ul></li> </ul>
 43 
 44 */

 50 /** ProtoPopup provides a button-less info box with header, body and
 51     footer fields.  Since it has no buttons, closing is done via the
 52     ESC key.
 53     @class
 54     @constructor
 55     @requires prototype-1.6.3.js -- the wellknown Ajax library, as well as Scriptaculous' effects.js and dragdrop.js
 56     @param {STRING} id A unique string identifying a popup
 57     @param {OBJECT} config The configuration object
 58     @return ProtoPopup object
 59     @type object
 60 
 61 
 62 
 63     @example
 64     var info = new ProtoPopup('info', {
 65         header     : new Element('div').update('Info header'),
 66         footer     : 'Info footer',
 67         appendBody : true
 68     });
 69 
 70     // Show the first info
 71     info.setHtml({body : 'Info 1'}).show();
 72 
 73     // Append a second info in bold
 74     info.setHtml({body : new Element('strong').update('Info 2')}).show();
 75 */
 76 var ProtoPopup = Class.create(/** @lends ProtoPopup.prototype */{
 77 
 78     /** @ignore */
 79     initialize : function (id, config) {
 80 
 81         /**
 82            The ID of the popup's object and DIV.
 83          */
 84         this.id = id;
 85 
 86         /**
 87            The default configuration<br/>
 88 
 89            <div style="padding-left: 20px">
 90 
 91            <b>hideOnEscape</b> {BOOL} - If true hides the popup when
 92            pressing the ESC key. Defaults to true.<br/>
 93 
 94            <b>centerOnCreation</b> {BOOL} - Center the popup relative
 95            to the viewport when creating it. Lateron it will be placed
 96            where the user drags it to. The position is remembered
 97            across hiding/showing until a full page reload
 98            occurs. Defaults to true.<br/>
 99 
100            <b>appendHeader</b> {BOOL} - If true, calls to {@link #setHtml}
101            will append to the popup's header section.  Defaults to true.<br/>
102 
103            <b>appendBody</b> {BOOL} - If true, calls to {@link #setHtml}
104            will append to the popup's body section.  Defaults to true.<br/>
105 
106            <b>appendFooter</b> {BOOL} - If true, calls to {@link #setHtml}
107            will append to the popup's footer section. Defaults to true.<br/>
108 
109            <b>header</b> {STRING} - The initial value of the header
110            section. Defaults to undefined.<br/>
111 
112            <b>body</b> {STRING} - The initial value of the footer
113            section. Defaults to undefined.<br/>
114 
115            <b>footer</b> {STRING} - The initial value of the body
116            section. Defaults to undefined.<br/>
117 
118            <b>width</b> {STRING} - The width of the popup. Overwrites the
119            width set in a CSS file. Defaults to undefined.<br/>
120 
121            <b>minWidth</b> {STRING} - The minimum width of the popup. Overwrites the
122            min-width set in a CSS file. Defaults to undefined.<br/>
123 
124            <b>maxWidth</b> {STRING} - The maximum width of the popup. Overwrites the
125            max-width set in a CSS file. Defaults to undefined.<br/>
126 
127            <b>height</b> {STRING} - The height of the popup. Overwrites the
128            height set in a CSS file. Defaults to undefined.<br/>
129 
130            <b>minHeight</b> {STRING} - The minimum height of the popup. Overwrites the
131            min-height set in a CSS file. Defaults to undefined.<br/>
132 
133            <b>maxHeight</b> {STRING} - The maximum height of the popup. Overwrites the
134            max-height set in a CSS file. Defaults to undefined.<br/>
135 
136            <b>cancelIconSrc</b> {STRING} - The src URL of the cancel
137            icon. Defaults to 'images/cancel.png'<br/>
138 
139            <b>zIndex</b> {NUMBER} - The z-index of the popup. Defaults
140            to 1.<br/>
141 
142            <b>modal</b> {BOOL} - If true, the popup will be of the
143            'modal' kind, the document being greyed out. The modal
144            overlay will have a z-index of 1 less than the popup's
145            z-index. Defaults to false.<br/>
146 
147            <b>opacity</b> {NUMBER} - The opacity of the overlay
148            greying out the screen in the case of a 'modal' popup (see
149            above). Defaults to 0.6.<br/>
150 
151            <b>headerBackgroundImage</b> {STRING} - CSS property
152            'background-image' for the header section. Defaults to
153            undefined.<br/>
154 
155            <b>bodyBackgroundImage</b> {STRING} - CSS property
156            'background-image' for the body section. Defaults to
157            undefined.<br/>
158 
159            <b>footerBackgroundImage</b> {STRING} - CSS property
160            'background-image' for the footer section. Defaults to
161            undefined.<br/>
162 
163            <b>draggableOptions</b> {OBJECT} - Options passed down to
164            underlying Scriptaculous' Draggable object. See
165            http://wiki.github.com/madrobby/scriptaculous/draggable for
166            more information.<br/>
167 
168            </div>
169 
170         */
171         this.config = {
172             hideOnEscape     : true,
173             centerOnCreation : true,            
174             appendHeader     : false,
175             appendBody       : false,
176             appendFooter     : false,
177             header           : undefined,
178             body             : undefined,
179             footer           : undefined,
180             width            : undefined,
181             minWidth         : undefined,
182             maxWidth         : undefined,
183             height           : undefined,
184             minHeight        : undefined,
185             maxHeight        : undefined,
186             cancelIconSrc    : 'images/cancel.png',
187             zIndex           : 1,
188             modal            : false,
189             opacity          : .6,
190             headerBackgroundImage : undefined,
191             bodyBackgroundImage   : undefined,
192             footerBackgroundImage : undefined,
193             draggableOptions      : {
194                 zindex: 1
195             }
196         };
197 
198         // merge in the config
199         Object.extend(this.config, (config || {}));
200 
201         // pass z-index to draggable
202         if (this.config.zIndex) {
203             this.config.draggableOptions.zindex = this.config.zIndex;
204         }
205 
206         // maybe attach keyUp handler to hide via ESC
207         if (this.config.hideOnEscape) {
208             Event.observe(document, 'keyup', this.hideOnEscape);
209         }
210 
211         var popup = new Element('div', {id : id, 'class' : 'proto-popup'})
212             .setStyle({zIndex : this.config.zIndex}).hide();
213         if (this.config.width)     popup.setStyle({width     : this.config.width});
214         if (this.config.minWidth)  popup.setStyle({minWidth  : this.config.minWidth});
215         if (this.config.maxWidth)  popup.setStyle({maxWidth  : this.config.maxWidth});
216         if (this.config.height)    popup.setStyle({height    : this.config.height});
217         if (this.config.minHeight) popup.setStyle({minHeight : this.config.minHeight});
218         if (this.config.maxHeight) popup.setStyle({maxHeight : this.config.maxHeight});
219 
220         /**
221            The popup div having {@link #id} as its ID, 'proto-popup'
222            as its class attribute.
223          */
224         this.popup = popup;
225 
226         // insert the cancel icon and attach click handler
227         if (this.config.cancelIconSrc) {
228             popup.insert(new Element('img', {
229                 id      : popup.id+'-cancel-button',
230                 src     : this.config.cancelIconSrc,
231                 'class' : 'proto-popup-cancel'
232             }).observe('click', function(e) { this.hide(); Event.stop(e) }.bind(this)));;
233         }
234 
235         /**
236            Array holding the popup's section names 'header', 'body' and 'footer'.
237          */
238         this.sections = ['header', 'body', 'footer'];
239 
240         /**
241            The popup's 'header' section is the popup's first DOM
242            child. It's ID is "id+'-header'", it's class attribute
243            'proto-popup-header'.   It's content is set via
244            {@link #setHtml}.
245 
246            @name ProtoPopup.prototype.header
247 
248          */
249         /**
250            The popup's 'body' section is the popup's first DOM
251            child. It's ID is "id+'-body'", it's class attribute
252            'proto-popup-body'.  It's content is set via
253            {@link #setHtml}.
254 
255            @name ProtoPopup.prototype.body
256 
257          */
258         /**
259            The popup's 'footer' section is the popup's first DOM
260            child. It's ID is "id+'-footer'", it's class attribute
261            'proto-popup-footer'.  It's content is set via
262            {@link #setHtml}.
263 
264            @name ProtoPopup.prototype.footer
265 
266          */
267         this.sections.each(function(section) {
268             var se = new Element('div', {id : id+'-'+section, 'class' : 'proto-popup-'+section}).hide();
269             if (this.config[section+'BackgroundImage']) {
270                 se.setStyle({backgroundImage: this.config[section+'BackgroundImage'],
271                              backgroundRepeat: 'no-repeat'});
272             }
273             this[section] = se;
274             popup.insert(se);
275         }.bind(this));
276 
277         // maybe add a modal overlay
278         var ie_opac = this.config.opacity * 100;
279         if (this.config.modal) {
280             var overlay = this.overlay = Element('div', {id: id+'-overlay', 'class': 'proto-popup-overlay'})
281                 .setStyle({opacity: this.config.opacity,
282                             filter: 'alpha(opacity='+ie_opac+')',
283                             zIndex: this.config.zIndex-1})
284                 .hide();
285             document.body.appendChild(overlay);
286         }
287 
288         // append to document body
289         document.body.appendChild(this.popup);
290 
291         // make it draggable
292         new Draggable(id, this.config.draggableOptions);
293         
294         // normally it's position:fixed, but IE6- does not understand it
295         if (Prototype.Browser.IEVersion < 7) {
296             popup.setStyle({position: 'absolute'});
297         } else {
298             popup.setStyle({position: 'fixed'});
299         }
300 
301         // set HTML
302         this.setHtml(this.config);
303 
304         // remember us
305         ProtoPopup.id2obj[id] = this;
306 
307         /**
308            Array of functions that will be executed at the end of the
309            popup initialization.
310          */
311         this.onShow = [];
312     },

314     /**
315        Fired when the ESC key is pressed and the config option
316        hideOnEscape is true (which is the default).
317        @event
318      */
319     hideOnEscape : function(event) {
320         if (event.keyCode == 27) { // KEY_ESC
321             Object.keys(ProtoPopup.id2obj).each(function(pp) {
322                     if (ProtoPopup.id2obj[pp].config.hideOnEscape) {
323                         ProtoPopup.id2obj[pp].hide();
324                     }
325             });
326         }
327

330     /**
331        Chainable instance method to set the HTML in any of the three popup
332        sections 'header', 'body' or 'footer'.
333 
334        @param config An object whose keys represent said three
335        sections.  The corresponding values can be plain text, HTML, a
336        DOM node or any object having a toHTML() method.
337 
338        @example
339        var info = new ProtoPopup('info');
340        info.setHtml({header : 'Info header', body : 'My Info'});
341     */
342     setHtml : function(config) {
343         this.sections.each(function(section) {
344                 if (config && config[section] != undefined) {
345                 if (!this.config['append'+section.capitalize()]) {
346                     this[section].innerHTML = '';
347                     
348                 }
349                 this[section].insert(config[section]).show();
350             }
351         }.bind(this));
352         return this;
353     },
354 
355     /**
356        Chainable instance method to show the popup.
357        @example
358        var info = new ProtoPopup('info', {header : 'Info header'});
359        info.setHtml({body : 'My Info'}).show();
360     */
361     show : function() {
362         this.popup.show();
363 
364         if (this.config.centerOnCreation) {
365             this.centerIt();
366             this.config.centerOnCreation = false;
367         }
368 
369         // show section even when directly setting them via Prototype's Ajax.Updater
370         this.sections.each(function(section) {
371             var s = $(this.id + '-' + section);
372             if (s && s.innerHTML != '') { s.show(); }
373         }.bind(this));
374 
375         // maybe show modal
376         if (this.overlay) {
377             this.overlay.show();
378             var dim    = document.viewport.getDimensions();
379             var scroll = document.viewport.getScrollOffsets(); 
380             this.overlay.setStyle({height: dim.height+'px', top:  scroll.top+'px',
381                         width:  dim.width+'px',  left: scroll.left+'px'});
382             try {
383                 var html = $(document.getElementsByTagName('html')[0]);
384                 html.setStyle({overflow: 'hidden'});
385             } catch(er) {
386                 try {
387                     var body = $(document.getElementsByTagName('body')[0]);
388                     body.setStyle({overflow: 'hidden'});
389                 } catch(er) {}
390             }
391         }
392 
393         this.onShow.each(function(f) {
394             if (Object.isFunction(f)) f.defer();
395         });
396         return this;
397     },
398 
399     /**
400        Chainable instance method to hide the popup.
401        @example
402        var info = new ProtoPopup('info', {header : 'Info header'});
403        info.setHtml({body : 'My Info'}).show();
404 
405        // hide 3 seconds later after showing it
406        setTimeout(function() {pinfo.hide()}, 3000);
407 
408     */
409     hide : function() {
410         if (this.overlay) {
411             try {
412                 var html = $(document.getElementsByTagName('html')[0]);
413                 html.setStyle({overflow: 'auto'});
414             } catch(er) {
415                 try {
416                     var body = $(document.getElementsByTagName('body')[0]);
417                     body.setStyle({overflow: 'auto'});
418                 } catch(er) {}
419             }
420             this.overlay.hide();
421         }
422         this.popup.hide();
423         return this;
424     },
425 
426     /**
427        Chainable instance method to center the popup horizontally and
428        vertically on the viewport.  Internally called by {@link #show}.
429     */
430     centerIt : function() {
431         var windowDim  = document.viewport.getDimensions();
432         var popupDim   = this.popup.getDimensions();
433 
434         if (Prototype.Browser.IE && Prototype.Browser.IEVersion < 7 ){
435             var scroll     = document.viewport.getScrollOffsets();
436             var scrollLeft = scroll.left;
437             var scrollTop  = scroll.top;
438         } else {
439             var scrollLeft = 0;
440             var scrollTop  = 0;
441         }
442 
443         centerX = Math.round(windowDim.width / 2) + scrollLeft
444             - (popupDim.width  / 2) + 'px';
445 
446         centerY = Math.round(windowDim.height/ 2) + scrollTop
447             - (popupDim.height / 2) + 'px';
448 
449         this.popup.setStyle({left: centerX, top: centerY});
450 
451         return this;
452     },

454     /**
455        Non-chainable instance method to create and return a button element.
456        This method is not used by ProtoPopup, but by its child classes
457        {@link ProtoPopup.Alert} and {@link ProtoPopup.Confirm}.<br><br>
458 
459        It's CSS class is "'proto-popup-'+name+'-btn'".
460 
461     */
462     makeButton : function(name, label) {
463         var btn = new Element('input', {
464             id      : this.id + '-' + name + '-btn',
465             type    : 'button',
466             value   : label,
467             'class' : 'proto-popup-btn'
468         });
469 
470         // maybe add a background image
471         if (this.config[name + 'BtnBackgroundImage']) {
472             btn.setStyle({backgroundImage: this.config[name + 'BtnBackgroundImage']});
473         }
474 
475         return btn;
476    }

479 /**
480    Class method returning get() methods for ProtoPopup and its
481    derivations.
482    @param newProtoPopup Function returning a constructor making ProtoPopup or
483    one of derivative objects.
484    @return get() method
485 */
486 ProtoPopup.makeGetFor = function(newProtoPopup) {
487     return function(id, config) {
488         var pp = ProtoPopup.id2obj[id];
489         if (pp) {
490             // we have it
491             pp.setHtml(config);
492             return pp;
493         }
494         // create it
495         pp = newProtoPopup(id, config);
496         return pp;
497     }
498 }

500 /**
501    Class method returning makeFunction() methods for ProtoPopup and its
502    derivations.
503    @param newProtoPopup Function returning a constructor making ProtoPopup or
504    one of derivative objects.
505    @return get() method
506 */
507 ProtoPopup.makeMakeFunction = function(newProtoPopup) {
508     return function(id, config) {
509         var pp;
510         if (Object.isUndefined(pp)) {
511             pp = newProtoPopup(id, config);
512             if (config && config.arg) {
513                 // attach original object
514                 config.arg.callee.obj = pp;
515             }
516         }
517         return function(msg) {
518             pp.setHtml({body : msg}).show();
519         }
520     }
521 }

523 /**
524    Class method returning (maybe first create) a draggable popup DIV
525    for info boxes.  Given the same id argument returns the same popup
526    object, following the singleton pattern.
527    @function
528    @param {STRING} id The ID of the popup.
529    @param {OBJECT} config The {@link #config} object.
530    @return The initialized and draggable popup.
531    @example Make a custom info() method serving as a convenient wrapper.
532 
533    The popup shown by info() will always have the string
534    'Newest Information' in the popup's header section, while the
535    argument passed to info() will be wrapped in a DIV that
536    will be inserted in the popup's body section.
537    
538         // make the custom info function
539         var info = function(msg) {
540             ProtoPopup.get('info', {
541                 header     : 'Info header',
542                 body       : new Element('div').update(msg),
543             });
544         }
545 
546         // first use
547         info('My first info');
548 
549         // second use (using the same popup, showing the same header)
550         info('My second info'); 
551 */
552 ProtoPopup.get = ProtoPopup.makeGetFor(function(id, config) {
553     return new ProtoPopup(id, config)
554 });
